public class Solution {
  public static final int N = 10000;

  public static void main(String[] args) {
    System.out.println("Calculating the running sum from 1 to " + N);
    calculateRunningSum();
  }

  public static void calculateRunningSum() {
    int runningSum = 0;

    for (int i = 0; i <= N; i++) {
      runningSum += i;
      System.out.println(runningSum);
    }
  }
}
