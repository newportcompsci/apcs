## Problem 11

```java
public static void main(String[] args) {
		for (int line = 1; line <= 4; line++) {
			int numDash = -line + 5;
			int numStar = 2*line;
			
			for (int i = 0; i < numDash; i++) {
				System.out.print("-");
			}
			
			for (int i = 0; i < numStar; i++) {
				System.out.print("*");
			}

			for (int i = 0; i < numDash; i++) {
				System.out.print("-");
			}
			
			System.out.println();
		}

		for (int line = 3; line > 0; line--) {
			int numDash = -line + 5;
			int numStar = 2*line;
			
			for (int i = 0; i < numDash; i++) {
				System.out.print("-");
			}
			
			for (int i = 0; i < numStar; i++) {
				System.out.print("*");
			}

			for (int i = 0; i < numDash; i++) {
				System.out.print("-");
			}
			
			System.out.println();
		}
	}

```
